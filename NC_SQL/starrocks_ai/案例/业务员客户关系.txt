select b.custcode,
       b.custname,
       b.pk_cubasdoc,
       m.pk_cumandoc,m.pk_resppsn1,
       m.sealflag, --是否封存
       sk.userid, --用户PK,
       sk.pk_psndoc, --业务员pk,
       sk.PSNCODE,
       sk.PSNNAME
  from  bd_cubasdoc b
  left  join bd_cumandoc m
    on b.pk_cubasdoc = m.pk_cubasdoc
   and m.pk_corp = '1006'
   and m.custflag in ('0', '2')
   and m.dr = 0
   and b.dr = 0
   and m.sealflag is null
  left join
  (select sk.*,p.PSNCODE PSNCODE,p.PSNNAME PSNNAME,p.pk_psndoc pkpsndoc from
  sm_userandclerk sk join bd_psndoc p on sk.pk_psndoc = p.pk_psnbasdoc
  and p.pk_corp = '1006' and p.dr = 0 ) sk
    on m.pk_resppsn1 = sk.pkpsndoc
   and sk.dr = 0;

select
    cum.PK_RESPDEPT1,cum.PK_RESPPSN1
from BD_CUBASDOC cub
left join BD_CUMANDOC cum on cub.PK_CUBASDOC=cum.PK_CUBASDOC
    and cum.PK_CORP='1006'
    and cum.DR=0
    and cub.DR=0
    and cum.SEALFLAG is null --封存
    and cub.SEALFLAG is null
    and cum.CUSTFLAG in ('0','2') --客户（排除了供应商）
where cub.CUSTCODE='18604599222';


select
    cub.CUSTNAME,cub.CUSTCODE,cub.SEALFLAG,cum.PK_RESPDEPT1,cum.PK_RESPPSN1,psn.PSNCODE,psn.PSNNAME,psn.PK_DEPTDOC,PK_PSNDOC
from BD_CUBASDOC cub
left join BD_CUMANDOC cum on cub.PK_CUBASDOC=cum.PK_CUBASDOC
    and cum.PK_CORP='1006'
    and cum.DR=0
    and cub.DR=0
    and cum.SEALFLAG is null --封存
    and cub.SEALFLAG is null
    and cum.CUSTFLAG in ('0','2') --客户（排除了供应商）
left join BD_PSNDOC psn on cum.PK_RESPPSN1=psn.PK_PSNDOC
    and psn.DR=0
    and psn.PK_CORP='1006'
where cub.CUSTCODE='18604599222';
